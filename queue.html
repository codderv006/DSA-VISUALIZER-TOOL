Procedure enqueue(data)


1. START
2. if queue is full
return overflow
3. elseif
rear ← rear + 1
queue[rear] ← data
return true
4. END

Procedure dequeue


1. START
2. if queue is empty
return underflow
3. elseif

return true
end procedure
5. END
